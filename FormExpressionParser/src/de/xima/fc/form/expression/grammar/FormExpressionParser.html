NON-TERMINALS

Template ::= ( ( ( LosChars ( LosOpen TemplateProgram )? ) | ( LosOpen TemplateProgram ) ) <EOF> ) 
/* Los = Literal Output Statement */
LosChars ::= ( ( <LosChar> )+ )
LosOpen ::= ( <LosOpen> )
LosBody ::= ( <LosBodyClose> ( <LosChar> )* ( <LosOpen> )? )
CompleteProgram ::= ( ( ( ClauseOrStatement )+ <EOF> ) | <EOF> )
TemplateProgram ::= ( ( ClauseOrStatement | LosBody )+ )
ClauseOrStatement ::= ( ForLoop | IfClause | FunctionClause | WhileLoop | DoWhileLoop | SwitchClause | TryClause | WithClause | ThrowClause | BreakClause | ContinueClause | ReturnClause | Statement )
FunctionClause ::= ( <Function> VariableWithOptionalScope <ParenOpen> ( ( IdentifierName )? ( <Comma> IdentifierName )* ) <ParenClose> ( Block ) )
WithClause ::= ( <With> <ParenOpen> IdentifierName ( <Comma> IdentifierName )* <ParenClose> BlockOrClause )
ContinueClause ::= ( <Continue> ( <Identifier> )? <SemiColon> )
BreakClause ::= ( <Break> ( <Identifier> )? <SemiColon> )
ReturnClause ::= ( <Return> ( Expression )? <SemiColon> )
ThrowClause ::= ( <Throw> Expression <SemiColon> )
SwitchClause ::= ( <Switch> ( <AngleOpen> <Identifier> <AngleClose> )? <ParenOpen> Expression <ParenClose> <BracesOpen> ( SwitchCase )* ( SwitchDefault )? <BracesClose> )
SwitchCase ::= ( ( <Case> Expression <Colon> )+ ( ClauseOrStatement )+ )
SwitchDefault ::= ( <Default> <Colon> ( ClauseOrStatement )+ )
TryClause ::= ( <Try> ( Block ) <Catch> <ParenOpen> <Identifier> <ParenClose> ( Block ) )
DoWhileLoop ::= ( <Do> ( <AngleOpen> <Identifier> <AngleClose> )? BlockOrClause <While> <ParenOpen> Expression <ParenClose> <SemiColon> )
WhileLoop ::= ( <While> ( <AngleOpen> <Identifier> <AngleClose> )? <ParenOpen> Expression <ParenClose> BlockOrClause )
ForLoop ::= ( <For> ( <AngleOpen> <Identifier> <AngleClose> )? <ParenOpen> ( ( <Identifier> <Colon> Expression ) | ( ExpressionListCommaSeparated <SemiColon> ExpressionListCommaSeparated <SemiColon> ExpressionListCommaSeparated ) ) <ParenClose> BlockOrClause )
IfClause ::= ( <If> <ParenOpen> Expression <ParenClose> BlockOrClause ( <Else> BlockOrClause )? )
BlockOrClause ::= Block
| ClauseOrStatement
Block ::= ( <BracesOpen> ( ( ClauseOrStatement | LosBody )* ) <BracesClose> )
ExpressionListCommaSeparated ::= ( ( Expression )? ( <Comma> Expression )* )
Statement ::= ( ( ( Expression ( <SemiColon> | LosBody ) ) ) | EmptyStatement )
EmptyStatement ::= ( <SemiColon> )
Expression ::= EqualTypeExpression
EqualTypeExpression ::= ( TernaryExpression ( ( <Equal> TernaryExpression ) | ( <PlusEqual> TernaryExpression ) | ( <DashEqual> TernaryExpression ) | ( <DoubleStarEqual> TernaryExpression ) | ( <StarEqual> TernaryExpression ) | ( <SlashEqual> TernaryExpression ) | ( <PercentEqual> TernaryExpression ) | ( <DoubleAngleOpenEqual> TernaryExpression ) | ( <DoubleAngleCloseEqual> TernaryExpression ) | ( <TripleAngleOpenEqual> TernaryExpression ) | ( <TripleAngleCloseEqual> TernaryExpression ) | ( <DoubleAmpersandEqual> TernaryExpression ) | ( <DoubleBarEqual> TernaryExpression ) | ( <AmpersandEqual> TernaryExpression ) | ( <CircumflexEqual> TernaryExpression ) | ( <BarEqual> TernaryExpression ) )* )
TernaryExpression ::= ( DoubleBarExpression ( <QuestionMark> DoubleBarExpression <Colon> DoubleBarExpression )? )
DoubleBarExpression ::= ( DoubleAmpersandExpression ( <DoubleBar> DoubleAmpersandExpression )* )
DoubleAmpersandExpression ::= ( BarExpression ( <DoubleAmpersand> BarExpression )* )
BarExpression ::= ( CircumflexExpression ( <Bar> CircumflexExpression )* )
CircumflexExpression ::= ( AmpersandExpression ( <Circumflex> AmpersandExpression )* )
AmpersandExpression ::= ( DoubleEqualTypeExpression ( <Ampersand> DoubleEqualTypeExpression )* )
DoubleEqualTypeExpression ::= ( AngleTypeExpression ( ( <ExclamationTilde> AngleTypeExpression ) | ( <EqualTilde> AngleTypeExpression ) | ( <TripleEqual> AngleTypeExpression ) | ( <DoubleEqual> AngleTypeExpression ) | ( <ExclamationDoubleEqual> AngleTypeExpression ) | ( <ExclamationEqual> AngleTypeExpression ) )* )
AngleTypeExpression ::= ( MultiAngleExpression ( ( <AngleOpen> MultiAngleExpression ) | ( <AngleClose> MultiAngleExpression ) | ( <AngleOpenEqual> MultiAngleExpression ) | ( <AngleCloseEqual> MultiAngleExpression ) )* )
MultiAngleExpression ::= ( PlusLikeExpression ( ( <DoubleAngleOpen> PlusLikeExpression ) | ( <DoubleAngleClose> PlusLikeExpression ) | ( <TripleAngleOpen> PlusLikeExpression ) | ( <TripleAngleClose> PlusLikeExpression ) )* )
PlusLikeExpression ::= ( StarLikeExpression ( ( <Plus> StarLikeExpression ) | ( <Dash> StarLikeExpression ) )* )
StarLikeExpression ::= DoubleStarExpression ( ( ( <Star> DoubleStarExpression ) | ( <Slash> DoubleStarExpression ) | ( <Percent> DoubleStarExpression ) )* )
DoubleStarExpression ::= ( UnaryExpression ( <DoubleStar> DoubleStarExpression )? )
UnaryExpression ::= ( ( <DoublePlus> UnaryExpression ) | ( <DoubleDash> UnaryExpression ) | ( <Plus> UnaryExpression ) | ( <Dash> UnaryExpression ) | ( <Exclamation> UnaryExpression ) | PropertyExpression )
PropertyExpression ::= ( FunctionTypeUnaryExpression ( <Dot> IdentifierNameAfterDot | <BracketOpen> Expression <BracketClose> | <ParenOpen> FunctionArgumentList <ParenClose> )* )
IdentifierNameAfterDot ::= ( <IdentifierAfterDot> )
IdentifierName ::= ( <Identifier> )
FunctionArgumentList ::= ( ( Expression )? ( <Comma> Expression )* )
FunctionTypeUnaryExpression ::= ( ( Literal | VariableWithOptionalScope | ( <ParenOpen> Expression <ParenClose> ) ) )
VariableWithOptionalScope ::= ( <Identifier> ( <ScopeSeparator> <Identifier> )? )
Number ::= ( <Integer> | <Float> )
String ::= ( <SingleQuotedString> | <DoubleQuotedString> )
Array ::= ( <BracketOpen> ( <BracketClose> | ( Expression ( <Comma> Expression )* <BracketClose> ) ) )
Hash ::= ( <BracesOpen> ( <BracesClose> | ( HashEntry ( <Comma> HashEntry )* <BracesClose> ) ) )
HashEntry ::= ( Literal | IdentifierName ) <Colon> Expression
Null ::= <Null>
Boolean ::= ( <True> | <False> )
Log ::= ( ( <LogInfo> | <LogError> | <LogDebug> | <LogWarn> ) <ParenOpen> Expression <ParenClose> )
Exception ::= ( <Exception> <ParenOpen> Expression <ParenClose> )
Function ::= ( <LambdaArrow> <ParenOpen> ( ( IdentifierName )? ( <Comma> IdentifierName )* ) <ParenClose> ( Block ) )
